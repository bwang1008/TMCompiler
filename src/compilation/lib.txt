// private functions
int div(int a, int b) {
	if(a == 0) {
		return 0;
	}
	
	if(a < 0 && b < 0) {
		return div(-a, -b);
	}
	if(a < 0) {
		return div(-a, b);
	}
	if(b < 0) {
		return div(a, -b);
	}
	// replace with enhanced binary search? then need division by 2...

	int ans = 0;
	while(b * (ans + 1) < a)  {
		ans += 1;
	}
	
	return ans;	
}

int mod(int a, int b) {
	int d = div(a, b);
	return a - b * d;
}
